Rodzaje pól

Rodzaj pola Opis Trzeba getter i setter?
public możemy się do niego dostać z każdego miejsca NIE
private możemy się do niego dostać tylko w obrębie klasy TAK

gettery i settery

Getter to funkcja, która zwraca nam wartość pola prywatnego
Setter najczęściej w argumencie przyjmuje wartość i ustawia ją jako wartość pola prywatnego
Używamy je m. in dlatego, żeby móc walidować dane wrzucane do pól

class nazwaKlasy{

private $nazwaPola;

function setNazwaPola($value){
$this->nazwaPola = $value;
}

function getNazwaPola($value){
return $this->nazwaPola;
}

}

__construct() i __destruct

__construct jest wywoływany przy tworzeniu klasy. W jego argumentach możemy podać wartości, które chcemy wpisać do pól
__destruct jest wywoływany przy usuwaniu obiektu

class nazwaKlasy{
public $nazwaPola;

public $nazwaPola2;

function __construct($wartosc1, $wartosc2){
$this->nazwaPola = $wartosc1;
$this->nazwaPola2 = $wartosc2;
}

function __destruct(){
echo "Zniszczono obiekt";
}
}
$obiekt = new nazwaKlasy("x", "y");
echo $obiekt->$nazwaPola."<br>"; // x
echo $obiekt->$nazwaPola2."<br>"; // y
unset($obiekt); // Zniszczono obiekt
